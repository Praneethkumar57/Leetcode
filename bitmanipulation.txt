// convert int to binary using right-shift operation

#include<bits/stdc++.h>
using namespace std;

string solve(){
  int n;
  cin>>n;
  string s="";
  if(n==0)return "0";
  while(n>0){
    s.push_back((n&1)?'1':'0');
    n >>=1;
  }
  reverse(s.begin(),s.end());
  return s;
}

int32_t main() 
{
    int t;
    cin>>t;
    while(t--){
      string s=solve();
      cout<<s<<endl;
    }
    return 0;
}
