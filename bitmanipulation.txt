// convert int to binary using right-shift operation

#include<bits/stdc++.h>
using namespace std;

string solve(){
  int n;
  cin>>n;
  string s="";
  if(n==0)return "0";
  while(n>0){
    s.push_back((n&1)?'1':'0');
    n >>=1;
  }
  reverse(s.begin(),s.end());
  return s;
}

int32_t main() 
{
    int t;
    cin>>t;
    while(t--){
      string s=solve();
      cout<<s<<endl;
    }
    return 0;
}


// convert bianry number to integer

#include<bits/stdc++.h>
using namespace std;

int solve(){
  int n;
  cin>>n;
  int k=1;
  int ans=0;
  string s=to_string(n);
  for(int i=s.size()-1;i>=0;i--){
    ans += (s[i]-'0')*k;
    k<<=1;
  }
  return ans;
}

int32_t main() 
{
    int t;
    cin>>t;
    while(t--){
      int s=solve();
      cout<<s<<endl;
    }
    return 0;
}

class Solution {
public:
    int totalHammingDistance(vector<int>& nums) {
        int n = nums.size();
        int ans = 0;
        for(int i = 0; i <= 30; i++){
            int c = 0;
            for(int elem : nums) if(elem & (1 << i)) c++;
            ans += c * (n - c);
        }
        return ans;
    }
};


class Solution {
public:
    vector<vector<int>> subsets(vector<int>& nums) {
        map<int,int> mpp;
        int n=nums.size();
        for(int i=0;i<n;i++){
            mpp[i]=nums[i];
        }
        int val = pow(2,n);
        vector<vector<int>> res;
        for(int i=0;i<val;i++){
            vector<int> t;
            int val = i;
            int ind=0;
            while(val>0){
                if(val&1)t.push_back(mpp[ind]);
                ind++;
                val>>=1;
            }
            res.push_back(t);
        }
        return res;
    }
};
